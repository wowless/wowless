macro(rock2c module rockspec)
  add_custom_command(
    OUTPUT ${module}.c
    COMMAND bootstraplua ${CMAKE_CURRENT_SOURCE_DIR}/tools/rock2c.lua ${rockspec}
    DEPENDS ${CMAKE_CURRENT_SOURCE_DIR}/tools/rock2c.lua ${rockspec}
  )
endmacro()
